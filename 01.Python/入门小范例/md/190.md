def random_lines_save(filename,gen_file_cnt=10):
    """
    随机选取文件的某些行并保存，想要生成这类文件的个数由参数
    @param: gen_file_cnt 指定 
    
    @param: filename 读入文件的完整路径
    @param: gen_file_cnt 想要产生的文件个数
    """
